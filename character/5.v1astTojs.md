# 第一步 astToOPCodes
生成通用的 compiler 代码，也就是 enviroment, 这一步将 ast 转为 opcodes。

看看 ID, string 的使用

```js
{

    // 每一个 ast 的元素都是 node, 在这里定义了 node.type 的函数，直接执行
    accept: function(node) {
      return this[node.type](node);
    },
    // HELPERS
    opcode: function(name) {
      this.opcodes.push({ opcode: name, args: [].slice.call(arguments, 1) });
    },
    ID: function(id) {
      this.addDepth(id.depth);
      // 确定 这个ID的 depth，如果是 ../../ 说明要去父对象
      this.opcode('getContext', id.depth);

      // 针对 parts 数组 [a,b,c] 依次取值。
      var name = id.parts[0];
      if (!name) {
        this.opcode('pushContext');
      } else {
        this.opcode('lookupOnContext', id.parts[0]);
      }

      for(var i=1, l=id.parts.length; i<l; i++) {
        this.opcode('lookup', id.parts[i]);
      }
    },
    // Handlebars.AST.StringNode = function(string) {
    //   this.type = "STRING";
    //   this.string = string;
    //   this.stringModeValue = string;
    // };  
    STRING: function(string) {
      this.opcode('pushString', string.string);
    },
}
```
# 第二步 OPCodesToJS
解码的 jsCompiler 会将 opcodes 数组里面的操作，转为 js 代码。

```js

{
    var opcodes = environment.opcodes, opcode;

    this.i = 0;

    for(l=opcodes.length; this.i<l; this.i++) {
        opcode = opcodes[this.i];

        if(opcode.opcode === 'DECLARE') {
            this[opcode.name] = opcode.value;
        } else {
            // 比如 pushString, lookup, lookupOnContext 方法
            this[opcode.opcode].apply(this, opcode.args);
        }
    }

    return this.createFunctionContext(asObject);
}


    // PUBLIC API: You can override these methods in a subclass to provide
    // alternative compiled forms for name lookup and buffering semantics
    nameLookup: function(parent, name /* , type*/) {
      if (/^[0-9]+$/.test(name)) {
        return parent + "[" + name + "]";
      } else if (JavaScriptCompiler.isValidJavaScriptVariableName(name)) {
        return parent + "." + name;
      }
      else {
        return parent + "['" + name + "']";
      }
    },
    
    // [lookupOnContext]
    //
    // On stack, before: ...
    // On stack, after: currentContext[name], ...
    //
    // Looks up the value of `name` on the current context and pushes
    // it onto the stack.
    lookupOnContext: function(name) {
      this.push(this.nameLookup('depth' + this.lastContext, name, 'context'));
    },

    // [pushString]
    //
    // On stack, before: ...
    // On stack, after: quotedString(string), ...
    //
    // Push a quoted version of `string` onto the stack
    pushString: function(string) {
      this.pushStackLiteral(this.quotedString(string));
    },

    pushStackLiteral: function(item) {
      return this.push(new Literal(item));
    },
    // [push]
    //
    // On stack, before: ...
    // On stack, after: expr, ...
    //
    // Push an expression onto the stack
    push: function(expr) {
      this.inlineStack.push(expr);
      return expr;
    },
```
